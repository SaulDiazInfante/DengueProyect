		\\
		\dot{S}_{-1} &= - \sigma B_2 S_{-1}
		\\
		\dot{Y}_{-1 c} &= (1 - \theta) \sigma B_2 S_{-1} - \alpha_c  Y_{-1c}
		\\
        \dot{Y}_{-1 h} &=  \theta \sigma B_2 S_{-1} - \alpha_h  Y_{-1h}
        \\
		\dot{R} & =
			\alpha_c( I_1 + I_2 + Y_{-1 c}) + \alpha_h Y_{-1 h}
	\end{aligned}
\end{equation}
